var maxArea = function(height) {
    let start = 0;
    let end  = height.length -1 ;
    let area =  0;
    while(start < end ){
        if(height[start] <= height[end] ){
            area = Math.max(area,height[start] * (end - start));
            start ++;
        }else{
            area = Math.max(area,height[end] * (end -start));
            end --;
        }
    }
    return area
 }
